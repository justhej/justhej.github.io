 1// 2.3.2
 2// v- 0

 4let expert = Expert()
 5
 6var gems = 0
 7func nav(){
 8    expert.moveForward()
 9    if expert.isOnGem{
10        expert.collectGem()
11        gems += 1
12    };
13    if expert.isBlocked{
14        if gems == 2{
15            expert.turnLockDown()
16        }
17        expert.turnLeft()
18        expert.turnLeft()
19    };
20}
21while gems < 6{
22    nav()
23    if gems == 1 || gems == 3 {
24        expert.moveForward()
25    }
26    if !expert.isBlockedLeft{
27        expert.turnLeft()
28        nav()
29    };
30    if !expert.isBlockedRight{
31        expert.turnRight()
32    };
33}
